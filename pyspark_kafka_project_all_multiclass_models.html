from pyspark.ml.feature import StringIndexer, VectorAssembler
from pyspark.ml import Pipeline
from pyspark.sql.functions import unix_timestamp

# use UNIX time so it can be an integer
training_df = training_df.withColumn(
    "unix_time", 
    unix_timestamp("time", "yyyy-MM-dd HH:mm:ss.SSS")
)

# Define all categorical fields to index
categorical_cols = ["ip", "method", "path", "referrer", "user_agent"]
indexers = [StringIndexer(inputCol=col, outputCol=col + "_idx", handleInvalid="keep") for col in categorical_cols]

# Update VectorAssembler to include it
assembler = VectorAssembler(
    inputCols=["status", "size", "unix_time"] + [col + "_idx" for col in ["ip", "method", "path", "referrer", "user_agent"]],
    outputCol="features"
)

# note that REQUEST_TYPE = method
# note that REQUEST ARGUMENT = path 

label_indexer =  StringIndexer(inputCol="category", outputCol="label")

print(label_indexer)

print(training_df.columns)

# Combine steps
pipeline = Pipeline(stages=indexers + [assembler, label_indexer])
pipeline_model = pipeline.fit(training_df)

label_indexer_model = pipeline_model.stages[-1]
print("Label -> Index Mapping:")
for i, label in enumerate(label_indexer_model.labels):
    print(f"{i}: {label}")

# Check for unmapped labels
actual_labels = training_df.select("category").distinct().rdd.flatMap(lambda x: x).collect()
mapped_labels = set(label_indexer_model.labels)
unmapped_labels = set(actual_labels) - mapped_labels
print("Unmapped Labels:", unmapped_labels)

print("Pipeline fit completed")
pipeline_model.save("/data/pipeline_model_MULTI5")
print("Pipeline saved")

log_transformed = pipeline_model.transform(training_df)

# show whole dataframe
log_transformed.show(5)

    0: "not anomalous",
    1: "slow_burst",
    2: "ddos",
    3: "error_spike",
    4: "cross_field",
    5: "false_positive",
    6: "size_anomaly"

def show_prediction_distribution(predictions_df, model_name):
    # Group by prediction and count
    pred_counts = predictions_df.groupBy("prediction").count().toPandas()
    
    # Map prediction index to class name
    pred_counts["prediction"] = pred_counts["prediction"].astype(int).map(label_mapping)
    
    # Sort for consistent ordering
    pred_counts = pred_counts.sort_values("prediction")
    
    print(f"\nPrediction Distribution for {model_name}:")
    print(pred_counts)
    
    # Optional bar plot
    pred_counts.set_index("prediction")["count"].plot(kind="bar", title=f"{model_name} Prediction Distribution", figsize=(8, 4))
    plt.ylabel("Count")
    plt.xlabel("Predicted Class")
    plt.xticks(rotation=45)
    plt.grid(axis="y", linestyle="--", alpha=0.7)
    plt.tight_layout()
    plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from pyspark.ml.evaluation import MulticlassClassificationEvaluator
from pyspark.sql.functions import col

# Class label mapping
label_mapping = {
   0: "not anomalous",
    1: "slow_burst",
    2: "ddos",
    3: "error_spike",
    4: "cross_field",
    5: "false_positive",
    6: "size_anomaly"
}

# Store results
model_results = []

def evaluate_multiclass_model(predictions, model_name):
    # Evaluators
    evaluator_precision = MulticlassClassificationEvaluator(labelCol="label", predictionCol="prediction", metricName="weightedPrecision")
    evaluator_recall = MulticlassClassificationEvaluator(labelCol="label", predictionCol="prediction", metricName="weightedRecall")
    evaluator_f1 = MulticlassClassificationEvaluator(labelCol="label", predictionCol="prediction", metricName="f1")
    evaluator_accuracy = MulticlassClassificationEvaluator(labelCol="label", predictionCol="prediction", metricName="accuracy")

    # Compute metrics
    accuracy = evaluator_accuracy.evaluate(predictions)
    precision = evaluator_precision.evaluate(predictions)
    recall = evaluator_recall.evaluate(predictions)
    f1 = evaluator_f1.evaluate(predictions)

    print(f"{model_name} Metrics:")
    print(f"Accuracy:  {accuracy:.3f}")
    print(f"Precision: {precision:.3f}")
    print(f"Recall:    {recall:.3f}")
    print(f"F1 Score:  {f1:.3f}\n")

    model_results.append({
        "Model": model_name,
        "Accuracy": accuracy,
        "Precision": precision,
        "Recall": recall,
        "F1 Score": f1
    })

# Evaluate your models
evaluate_multiclass_model(rf_predictions, "Random Forest")
evaluate_multiclass_model(lr_predictions, "Logistic Regression")
# evaluate_multiclass_model(gbt_predictions, "Gradient Boosted Tree")
evaluate_multiclass_model(mlpc_predictions, "MLP Classifier")

# Convert to DataFrame
results_df = pd.DataFrame(model_results)
results_df.set_index("Model", inplace=True)

# Plot: Models as bars, metrics on x-axis
results_df.T.plot(kind="bar", figsize=(10, 6))
plt.title("Multiclass Model Performance Comparison")
plt.xlabel("Metrics")
plt.ylabel("Score")
plt.ylim(0, 1.05)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.xticks(rotation=0)
plt.legend(title="Model", loc="upper right")
plt.tight_layout()
plt.show()

kafka_timestamp_format = "dd/MMM/yyyy:HH:mm:ss Z"

# Convert date_time from string to timestamp using the correct format

    "event_time",
    coalesce(
        to_timestamp(col("date_time"), kafka_timestamp_format),
    )
)

# create windows

# adjust names to match the dataset used for training

    "ip_address as ip",
    "request_type as method",
    "request_arg as path",
    "status_code as status",
    "response_size as size",
    "referrer",
    "user_agent",
    "event_time"
)

# convert data to ints and calculate unix_time
renamed_df = renamed_df \
    .withColumn("status", coalesce(col("status").cast("int"), lit(0))) \
    .withColumn("size", coalesce(col("size").cast("int"), lit(0))) \
    .withColumn("unix_time", unix_timestamp("event_time"))

# Filter again for safety before applying models, ensuring unix_time is not null
